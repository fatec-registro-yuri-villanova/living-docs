                    containerColor = MaterialTheme.colorScheme.primary
                ),
                scrollBehavior = scrollBehavior
            )

                onClick = { /* Action */ },

                    .border(1.dp, MaterialTheme.colorScheme.primary, RoundedCornerShape(12.dp))
                    .padding(16.dp)

                        text = "O Poder do Modifier",

:::caution A ordem dos Modifiers importa!
O Modifier √© encadeado ‚Äî cada `.` adiciona comportamento **na ordem**. Compare:
- `.padding(16.dp).background(Color.Red)` ‚Üí padding **fora** do fundo

:::
